* create cordova project
* add android platform
* add plugin using below command
	cordova plugin add <location of PluginLocation folder>

for example
	cordova plugin add /Users/ashwinichandel/Desktop/PluginLocation 

* open your CordovaApp.java(this naming of class me be difference in project), for this will extended by CordovaActivity class, add below lines in this file

			   @Override
				public Object onMessage(String arg0, Object arg1) {
					if (arg0.equals("onPageFinished")) {
						System.err.println("onPageFinished");
						if(getIntent().getAction().equalsIgnoreCase("from_notification_bar")) {
				        	System.out.println("from_notification_bar !!!!!!!!!!");
				        	Toast.makeText(this, "Successfully stoped", Toast.LENGTH_SHORT).show();
				        	Intent serviceintent = new Intent(this, BackendService.class);
							serviceintent.putExtra("key", "stop");
							startService(serviceintent);
				       }
					}
					return super.onMessage(arg0, arg1);
				}

* now you can build or run your project
